syms z; % Se define a z como una variable simbólica
% Función f(z)
func1 = z^2 + 5i*z + 3 - 1i;

% Punto z0
z0 = 6 - 1i;

% Derivada de f'(z)
dfunc1 = diff(func1, z);

% Derivada en el punto z0
valor_dfunc1 = subs(dfunc1, z, z0);

% Mostramos el resultado
disp(['La derivada f''(z) evaluada en z0 = ', num2str(z0), ' es:']);
disp(valor_dfunc1);


func2 = (z^2 + 2i*z) / (z - 1i);

% Definimos el punto z0
z0 = 4 + 2i;

% Calculamos la derivada de f(z)
df = diff(func2, z);

% Evaluamos la derivada en el punto z0
df_value = subs(df, z, z0);

% Mostramos el resultado
disp(['La derivada de f(z) evaluada en z0 = ', num2str(z0), ' es:']);
disp(df_value);
